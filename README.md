## Challenge 3 - simple calculator

``` javascript
function calculate(num1, operation, num2) {
  switch (operation){
      case "+":
        var x = num1+num2;
        return x;
      break;
      case "-":
      var x = num1-num2;
        return x;
      break;
      case "/":
        if (num2 == 0){
          return null; 
        } else 
          return num1/num2;
      break;
      case "*":
        return num1*num2;
      break;

        return null;

        }
 ``` 
